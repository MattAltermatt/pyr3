# Hardening Milestone (#71 + #70) Implementation Plan

> **For agentic workers:** Use `superpowers:subagent-driven-development` or
> `superpowers:executing-plans`. Step boxes use `- [ ]`.

**Goal:** Close the two open issues in the Hardening milestone — wire the BE
parity rig into CI (#71) and extract `loadInFlight` from `main.ts`'s closure
into a tested sequencer module (#70).

**Architecture:**
- **#71** adds a separate GitHub Actions job `verify-parity` to `.github/workflows/ci.yml`
  that runs `npm run test:parity` on `ubuntu-latest`. The 25-fixture rig spawns
  `bin/pyr3-render.ts` (Dawn `webgpu` npm) per fixture; on Linux Dawn picks up
  the SwiftShader Vulkan ICD that ships with `mesa-vulkan-drivers`. Job gates on
  `push: branches: [main]` + `workflow_dispatch` only — keeps PR cycles fast
  while still failing the post-merge run if tier-1 parity regresses.
- **#70** extracts the `loadInFlight` + `loadHookQueue` + `corpusQueue` +
  `navLocked` state from inside `createApp()` into a new `src/load-sequencer.ts`
  module exporting a `createLoadSequencer({ device, loadFromFile })` factory.
  `main.ts` constructs one and wires the same surfaces (`__pyr3LoadFlame`,
  `enqueueCorpus`, `navigateCorpus`) to it. Pure-logic unit tests cover the
  four sequencing risks the issue names, using `vi.useFakeTimers()` and a fake
  device whose `queue.onSubmittedWorkDone()` is a controllable Promise.

**Tech Stack:** GitHub Actions, Ubuntu runner, Dawn `webgpu` npm, Vitest +
fake timers, TypeScript.

- [ ] **Step 3: Write the minimal `src/load-sequencer.ts`:**

```ts
export interface LoadSequencerDeps {
  device: GPUDevice;
  loadFromFile: (file: File) => Promise<void>;
}

export interface LoadSequencer {
  loadFile(file: File): Promise<void>;
  enqueueHook(text: string, label?: string): Promise<void>;
  enqueueCorpus(load: () => Promise<void>): Promise<void>;
  tryNavigateCorpus(load: () => Promise<void>, isRenderInFlight: () => boolean): void;
  readonly inFlight: () => boolean;
}

export function createLoadSequencer({ device, loadFromFile }: LoadSequencerDeps): LoadSequencer {
  let loadInFlight = false;
  let loadHookQueue: Promise<void> = Promise.resolve();
  let corpusQueue: Promise<void> = Promise.resolve();
  let navLocked = false;

  const loadFile = async (file: File): Promise<void> => {
    if (loadInFlight) return;
    loadInFlight = true;
    try {
      await loadFromFile(file);
      await device.queue.onSubmittedWorkDone();
    } finally {
      loadInFlight = false;
    }
  };

  const enqueueHook = (text: string, label = 'test.flame'): Promise<void> => {
    const next = loadHookQueue.then(async () => {
      while (loadInFlight) await new Promise((r) => setTimeout(r, 25));
      await loadFile(new File([text], label, { type: 'text/xml' }));
    });
    loadHookQueue = next.catch(() => {});
    return next;
  };

  const enqueueCorpus = (load: () => Promise<void>): Promise<void> => {
    const next = corpusQueue.then(load);
    corpusQueue = next.catch(() => {});
    return next;
  };

  const tryNavigateCorpus = (load: () => Promise<void>, isRenderInFlight: () => boolean): void => {
    if (navLocked || isRenderInFlight()) return;
    navLocked = true;
    void enqueueCorpus(load).finally(() => { navLocked = false; });
  };

  return {
    loadFile,
    enqueueHook,
    enqueueCorpus,
    tryNavigateCorpus,
    inFlight: () => loadInFlight,
  };
}
```

```ts
it('serializes back-to-back __pyr3LoadFlame calls (no overlap)', async () => {
  const order: string[] = [];
  let releaseA: (() => void) | null = null;
  let releaseB: (() => void) | null = null;
  const loadFromFile = vi.fn(async (f: File) => {
    order.push(`start:${f.name}`);
    await new Promise<void>((r) => {
      if (f.name === 'a.flame') releaseA = r; else releaseB = r;
    });
    order.push(`end:${f.name}`);
  });
  const seq = createLoadSequencer({
    device: fakeDevice(() => Promise.resolve()),
    loadFromFile,
  });
  const pA = seq.enqueueHook('xmlA', 'a.flame');
  const pB = seq.enqueueHook('xmlB', 'b.flame');
  await Promise.resolve(); // let queue tick
  // Only A started
  expect(order).toEqual(['start:a.flame']);
  releaseA!();
  await pA;
  // After A resolves, B can start
  await new Promise((r) => setTimeout(r, 30));
  expect(order).toEqual(['start:a.flame', 'end:a.flame', 'start:b.flame']);
  releaseB!();
  await pB;
});
```

```ts
it('awaits device.queue.onSubmittedWorkDone before releasing loadInFlight', async () => {
  const events: string[] = [];
  let releaseDrain: (() => void) | null = null;
  const drain = () => new Promise<void>((r) => {
    events.push('drain-start');

  releaseDrain!();
  await p;
  expect(seq.inFlight()).toBe(false);
```
